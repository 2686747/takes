24  package org.takes.facets.auth.codecs;
25  
26  import java.io.ByteArrayInputStream;
27  import java.io.ByteArrayOutputStream;
28  import java.io.DataInputStream;
29  import java.io.DataOutputStream;
30  import java.io.IOException;
31  import java.util.Map;
32  import java.util.concurrent.ConcurrentHashMap;
33  import java.util.concurrent.ConcurrentMap;
34  import lombok.EqualsAndHashCode;
35  import org.takes.facets.auth.Identity;
36  
37  /**
38   * Compact codec.
39   *
40   * <p>The class is immutable and thread-safe.
41   *
42   * @author Yegor Bugayenko (yegor@teamed.io)
43   * @version $Id: b4521cc1e9c137f39c5bcc51c6e9035c04d5530e $
44   * @since 0.5
45   */
46  @EqualsAndHashCode
47  public final class CcCompact implements Codec {
48  
49      @Override
50      public byte[] encode(final Identity identity) throws IOException {
51          final ByteArrayOutputStream data = new ByteArrayOutputStream();
52          final DataOutputStream stream = new DataOutputStream(data);
53          try {
54              stream.writeUTF(identity.urn());
55              for (final Map.Entry<String, String> ent
56                  : identity.properties().entrySet()) {
57                  stream.writeUTF(ent.getKey());
58                  stream.writeUTF(ent.getValue());
59              }
60          } catch (final IOException ex) {
61              throw new IllegalArgumentException(ex);
62          } finally {
63              stream.close();
64          }
65          return data.toByteArray();
66      }
67  
68      @Override
69      public Identity decode(final byte[] bytes) throws IOException {
70          final ConcurrentMap<String, String> map =
71              new ConcurrentHashMap<String, String>(0);
72          final DataInputStream stream = new DataInputStream(
73              new ByteArrayInputStream(bytes)
74          );
75          try {
76              final String urn = stream.readUTF();
77              while (stream.available() > 0) {
78                  map.put(stream.readUTF(), stream.readUTF());
79              }
80              return new Identity.Simple(urn, map);
81          } catch (final IOException ex) {
82              throw new DecodingException(ex);
83          } finally {
84              stream.close();
85          }
86      }
87  
88  }
