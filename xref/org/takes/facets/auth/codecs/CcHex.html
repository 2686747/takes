24  package org.takes.facets.auth.codecs;
25  
26  import java.io.IOException;
27  import java.util.Arrays;
28  import lombok.EqualsAndHashCode;
29  import org.takes.facets.auth.Identity;
30  
31  /**
32   * Hex codec.
33   *
34   * <p>The class is immutable and thread-safe.
35   *
36   * @author Yegor Bugayenko (yegor@teamed.io)
37   * @version $Id$
38   * @since 0.1
39   */
40  @EqualsAndHashCode(of = "origin")
41  public final class CcHex implements Codec {
42  
43      /**
44       * Backward mapping table.
45       */
46      private static final byte[] BACK = {
47          -1, -1, -1, -1, -1, -1, -1, -1,
48          -1, -1, -1, -1, -1, -1, -1, -1,
49          -1, -1, -1, -1, -1, -1, -1, -1,
50          -1, -1, -1, -1, -1, -1, -1, -1,
51          -1, -1, -1, -1, -1, -1, -1, -1,
52          -1, -1, -1, -1, -1, -1, -1, -1,
53          0, 1, 2, 3, 4, 5, 6, 7,
54          8, 9, -1, -1, -1, -1, -1, -1,
55          -1, 0xa, 0xb, 0xc, 0xd, 0xe, 0xf,
56      };
57  
58      /**
59       * Forward mapping table.
60       */
61      private static final byte[] FWD = {
62          '0', '1', '2', '3', '4', '5', '6', '7',
63          '8', '9', 'A', 'B', 'C', 'D', 'E', 'F',
64      };
65  
66      /**
67       * Original codec.
68       */
69      private final transient Codec origin;
70  
71      /**
72       * Ctor.
73       * @param codec Original codec
74       */
75      public CcHex(final Codec codec) {
76          this.origin = codec;
77      }
78  
79      @Override
80      public byte[] encode(final Identity identity) throws IOException {
81          final byte[] raw = this.origin.encode(identity);
82          final byte[] out = new byte[raw.length << 1];
83          for (int idx = 0; idx < raw.length; ++idx) {
84              out[idx << 1] = CcHex.FWD[raw[idx] >> 4 & 0x0f];
85              out[(idx << 1) + 1] = CcHex.FWD[raw[idx] & 0x0f];
86          }
87          return out;
88      }
89  
90      @Override
91      public Identity decode(final byte[] bytes) throws IOException {
92          if ((bytes.length & 0x01) == 1) {
93              throw new DecodingException(
94                  String.format(
95                      "hex text must contain even number of chars: %s",
96                      Arrays.toString(bytes)
97                  )
98              );
99          }
100         final byte[] out = new byte[bytes.length >> 1];
101         for (int idx = 0; idx < out.length; ++idx) {
102             final int high = bytes[idx << 1];
103             final int low = bytes[(idx << 1) + 1];
104             out[idx] = (byte) ((CcHex.decode(high) << 4)
105                 + CcHex.decode(low));
106         }
107         return this.origin.decode(out);
108     }
109 
110     /**
111      * Convert hex to number.
112      * @param hex Hex number
113      * @return Decoded
114      */
115     private static int decode(final int hex) {
116         if (hex >= CcHex.BACK.length) {
117             throw new DecodingException(
118                 String.format("invalid hex char: 0x%2x", hex)
119             );
120         }
121         final int dec = CcHex.BACK[hex];
122         if (dec < 0) {
123             throw new DecodingException(
124                 String.format("invalid hex character: 0x%2x", hex)
125             );
126         }
127         return dec;
128     }
