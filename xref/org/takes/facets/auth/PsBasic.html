45  /**
46   * Pass that checks the user according RFC-2617.
47   *
48   * <p>The class is immutable and thread-safe.
49   *
50   * @author Endrigo Antonini (teamed@endrigo.com.br)
51   * @version $Id: 6a8c25c11d0636cb77dda7304590acff9ac1958e $
52   * @since 0.20
53   */
54  @EqualsAndHashCode(of = { "entry", "realm" })
55  public final class PsBasic implements Pass {
56  
57      /**
58       * Authorization response HTTP head.
59       */
60      private static final String AUTH_HEAD = "Basic";
61  
62      /**
63       * Entry to validate user information.
64       */
65      private final transient PsBasic.Entry entry;
66  
67      /**
68       * Realm.
69       */
70      private final transient String realm;
71  
72      /**
73       * Ctor.
74       * @param rlm Realm
75       * @param basic Entry
76       */
77      public PsBasic(final String rlm, final PsBasic.Entry basic) {
78          this.realm = rlm;
79          this.entry = basic;
80      }
81  
82      @Override
83      public Opt<Identity> enter(final Request request) throws IOException {
84          final String decoded = new String(
85              DatatypeConverter.parseBase64Binary(
86                  new RqHeaders.Smart(
87                      new RqHeaders.Base(request)
88                  ).single("authorization").split(AUTH_HEAD)[1]
89              )
90          ).trim();
91          final String user = decoded.split(":")[0];
92          final Opt<Identity> identity = this.entry.enter(
93              user,
94              decoded.substring(user.length() + 1)
95          );
96          if (!identity.has()) {
97              throw new RsForward(
98                  new RsWithHeader(
99                      new RsFlash("access denied", Level.WARNING),
100                     String.format(
101                         "WWW-Authenticate: Basic ream=\"%s\"",
102                         this.realm
103                     )
104                 ),
105                 HttpURLConnection.HTTP_UNAUTHORIZED,
106                 new RqHref.Base(request).href()
107             );
108         }
109         return identity;
110     }
111 
112     @Override
113     public Response exit(final Response response, final Identity identity)
114         throws IOException {
115         return response;
116     }

118     /**
119      * Entry interface that is used to check if the received information is
120      * valid.
121      *
122      * @author Endrigo Antonini (teamed@endrigo.com.br)
123      * @version $Id: 6a8c25c11d0636cb77dda7304590acff9ac1958e $
124      * @since 0.20
125      */
126     public interface Entry {
127 
128         /**
129          * Check if is a valid user.
130          * @param user User
131          * @param pwd Password
132          * @return Identity.
133          */
134         Opt<Identity> enter(String user, String pwd);
135     }
136 
137     /**
138      * Fake implementation of {@link PsBasic.Entry}.
139      *
140      * <p>The class is immutable and thread-safe.
141      *
142      * @author Endrigo Antonini (teamed@endrigo.com.br)
143      * @version $Id: 6a8c25c11d0636cb77dda7304590acff9ac1958e $
144      * @since 0.20
145      *
146      */
147     public static final class Fake implements PsBasic.Entry {
148 
149         /**
150          * Should we authenticate a user?
151          */
152         private final transient boolean condition;
153 
154         /**
155          * Ctor.
156          * @param cond Condition
157          */
158         public Fake(final boolean cond) {
159             this.condition = cond;
160         }
161 
162         @Override
163         public Opt<Identity> enter(final String usr, final String pwd) {
164             final Opt<Identity> user;
165             if (this.condition) {
166                 user = new Opt.Single<Identity>(
167                     new Identity.Simple(
168                         String.format("urn:basic:%s", usr)
169                     )
170                 );
171             } else {
172                 user = new Opt.Empty<Identity>();
173             }
174             return user;
175         }
176     }
177 
178     /**
179      * Empty check.
180      *
181      * @author Endrigo Antonini (teamed@endrigo.com.br)
182      * @version $Id: 6a8c25c11d0636cb77dda7304590acff9ac1958e $
183      * @since 0.20
184      */
185     public static final class Empty implements PsBasic.Entry {
186 
187         @Override
188         public Opt<Identity> enter(final String user, final String pwd) {
189             return new Opt.Empty<Identity>();
190         }
191     }
192 
193     /**
194      * Default entry.
195      *
196      * @author Georgy Vlasov (wlasowegor@gmail.com)
197      * @version $Id: 6a8c25c11d0636cb77dda7304590acff9ac1958e $
198      * @since 0.22
199      */
200     public static final class Default implements PsBasic.Entry {
201 
202         /**
203          * How keys in
204          * {@link org.takes.facets.auth.PsBasic.Default#usernames} are
205          * formatted.
206          */
207         private static final String KEY_FORMAT = "%s %s";
208 
209         /**
210          * Encoding for URLEncode#encode.
211          */
212         private static final String ENCODING = "UTF-8";
213 
214         /**
215          * Map from login/password pairs to URNs.
216          */
217         private final transient Map<String, String> usernames;
218 
219         /**
220          * Public ctor.
221          * @param users Strings with user's login, password and URN with
222          *  space characters as separators. Each of login, password and urn
223          *  are URL-encoded substrings. For example,
224          *  {@code "mike my%20password urn:jcabi-users:michael"}.
225          */
226         public Default(final String... users) {
227             this.usernames = new HashMap<String, String>(users.length);
228             for (final String user : users) {
229                 final String unified = user.replace("%20", "+");
230                 PsBasic.Default.validateUser(unified);
231                 this.usernames.put(
232                     PsBasic.Default.key(unified),
233                     unified.substring(unified.lastIndexOf(' ') + 1)
234                 );
235             }
236         }
237 
238         @Override
239         public Opt<Identity> enter(final String user, final String pwd) {
240             final Opt<String> urn = this.urn(user, pwd);
241             final Opt<Identity> identity;
242             if (urn.has()) {
243                 try {
244                     identity = new Opt.Single<Identity>(
245                         new Identity.Simple(
246                             URLDecoder.decode(urn.get(), Default.ENCODING)
247                         )
248                     );
249                 } catch (final UnsupportedEncodingException ex) {
250                     throw new IllegalStateException(ex);
251                 }
252             } else {
253                 identity = new Opt.Empty<Identity>();
254             }
255             return identity;
256         }
257 
258         /**
259          * Returns an URN corresponding to a login-password pair.
260          * @param user Login.
261          * @param pwd Password.
262          * @return Opt with URN or empty if there is no such login-password
263          *  pair.
264          */
265         private Opt<String> urn(final String user, final String pwd) {
266             final String urn;
267             try {
268                 urn = this.usernames.get(
269                     String.format(
270                         Default.KEY_FORMAT,
271                         URLEncoder.encode(
272                             user,
273                             Default.ENCODING
274                         ),
275                         URLEncoder.encode(
276                             pwd,
277                             Default.ENCODING
278                         )
279                     )
280                 );
281             } catch (final UnsupportedEncodingException ex) {
282                 throw new IllegalStateException(ex);
283             }
284             final Opt<String> opt;
285             if (urn == null) {
286                 opt = new Opt.Empty<String>();
287             } else {
288                 opt = new Opt.Single<String>(urn);
289             }
290             return opt;
291         }
292 
293         /**
294          * Creates a key for
295          *  {@link org.takes.facets.auth.PsBasic.Default#usernames} map.
296          * @param unified User string made of 3 urlencoded substrings
297          *  separated with non-urlencoded space characters.
298          * @return Login and password parts with <pre>%20</pre> replaced with
299          *  <pre>+</pre>.
300          */
301         private static String key(final String unified) {
302             return String.format(
303                 PsBasic.Default.KEY_FORMAT,
304                 unified.substring(0, unified.indexOf(' ')),
305                 unified.substring(
306                     unified.indexOf(' ') + 1,
307                     unified.lastIndexOf(' ')
308                 )
309             );
310         }
311 
312         /**
313          * Checks if a unified user string is correctly formatted.
314          * @param unified String with urlencoded user login, password and urn
315          *  separated with spaces.
316          */
317         private static void validateUser(final String unified) {
318             final boolean incorrectAmount = Default.countSpaces(unified) != 2;
319             final boolean nearby =
320                 unified.indexOf(' ') + 1 == unified.lastIndexOf(' ');
321             if (incorrectAmount || nearby) {
322                 throw new IllegalArgumentException(
323                     String.format(
324                         "One of users was incorrectly formatted: %s",
325                         unified
326                     )
327                 );
328             }
329         }
330 
331         /**
332          * Counts spaces in a string.
333          * @param string Any string.
334          * @return Amount of spaces in string.
335          */
336         private static int countSpaces(final String string) {
337             int spaces = 0;
338             for (final char character : string.toCharArray()) {
339                 if (character == ' ') {
340                     spaces += 1;
341                 }
342             }
343             return spaces;
344         }
345     }
