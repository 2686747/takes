24  package org.takes.facets.auth;
25  
26  import java.io.IOException;
27  import java.util.ArrayList;
28  import java.util.List;
29  import org.takes.Request;
30  import org.takes.Response;
31  import org.takes.misc.Opt;
32  
33  /**
34   * A Pass which you can enter only if you can enter every Pass in a list.
35   * @author Georgy Vlasov (wlasowegor@gmail.com)
36   * @version $Id: 8c76c10e9a09c8b0d4b2be89a2fe2ea641298117 $
37   * @since 0.22
38   */
39  public class PsAll implements Pass {
40  
41      /**
42       * Passes that have to be entered.
43       */
44      private final transient List<? extends Pass> all;
45  
46      /**
47       * Index of identity to return.
48       */
49      private final transient int index;
50  
51      /**
52       * Ctor.
53       * @param passes All Passes to be checked.
54       * @param identity Index of a Pass whose Identity to return on successful
55       *  {@link PsAll#enter(Request)}
56       */
57      public PsAll(final List<? extends Pass> passes, final int identity) {
58          this.all = new ArrayList<Pass>(passes);
59          this.index = identity;
60          if (this.index < 0) {
61              throw new IllegalArgumentException("Index must be >= 0");
62          }
63          if (this.index >= this.all.size()) {
64              throw new IllegalArgumentException(
65                  String.format(
66                      "Trying to return index %s from a list of %s passes",
67                      this.index,
68                      this.all.size()
69                  )
70              );
71          }
72      }
73  
74      @Override
75      public final Opt<Identity> enter(final Request request) throws IOException {
76          final Opt<Identity> result;
77          if (this.allMatch(request)) {
78              result = this.all.get(this.index).enter(request);
79          } else {
80              result = new Opt.Empty<Identity>();
81          }
82          return result;
83      }
84  
85      @Override
86      public final Response exit(final Response response, final Identity identity)
87          throws IOException {
88          if (this.index >= this.all.size()) {
89              throw new IOException(
90                  "Index of identity is greater than Pass collection size"
91              );
92          }
93          return this.all.get(this.index).exit(response, identity);
94      }
95  
96      /**
97       * Checks if you can enter every Pass with a request.
98       * @param request Request that is used to enter Passes.
99       * @return True if every request can be entered, false otherwise
100      * @throws IOException If any of enter attempts fail
101      */
102     private boolean allMatch(final Request request) throws IOException {
103         boolean success = true;
104         for (final Pass pass : this.all) {
105             if (!pass.enter(request).has()) {
106                 success = false;
107                 break;
108             }
109         }
110         return success;
111     }
112 }
