44  /**
45   * Request decorator that decodes FORM data from
46   * {@code multipart/form-data} format (RFC 2045).
47   *
48   * <p>For {@code } format use {@link org.takes.rq.RqForm}.
49   *
50   * <p>The class is immutable and thread-safe.
51   *
52   * @author Yegor Bugayenko (yegor@teamed.io)
53   * @version $Id$
54   * @since 0.9
55   * @link <a href="http://www.w3.org/TR/html401/interact/forms.html">Forms in HTML</a>
56   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
57   */
58  @EqualsAndHashCode(callSuper = true)
59  public final class RqMultipart extends RqWrap {
60  
61      /**
62       * Pattern to get boundary from header.
63       */
64      private static final Pattern BOUNDARY = Pattern.compile(
65          ".*[^a-z]boundary=([^;]+).*"
66      );
67  
68      /**
69       * Pattern to get name from header.
70       */
71      private static final Pattern NAME = Pattern.compile(
72          ".*[^a-z]name=\"([^\"]+)\".*"
73      );
74  
75      /**
76       * Map of params and values.
77       */
78      private final transient ConcurrentMap<String, List<Request>> map;
79  
80      /**
81       * Ctor.
82       * @param req Original request
83       * @throws IOException If fails
84       */
85      public RqMultipart(final Request req) throws IOException {
86          super(req);
87          final String header = new RqHeaders(req).header("Content-Type")
88              .iterator().next();
89          if (!header.toLowerCase(Locale.ENGLISH)
90              .startsWith("multipart/form-data")) {
91              throw new IOException(
92                  String.format(
93                      // @checkstyle LineLength (1 line)
94                      "RqMultipart can only parse multipart/form-data, while Content-Type specifies a different type: %s",
95                      header
96                  )
97              );
98          }
99          final Matcher matcher = RqMultipart.BOUNDARY.matcher(header);
100         if (!matcher.matches()) {
101             throw new IOException(
102                 String.format(
103                     "boundary is not specified in Content-Type header: %s",
104                     header
105                 )
106             );
107         }
108         final Collection<Request> requests = new LinkedList<Request>();
109         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
110         new RqPrint(req).printBody(baos);
111         final byte[] boundary = matcher.group(1).getBytes();
112         final byte[] body = baos.toByteArray();
113         int pos = 0;
114         while (pos < body.length) {
115             int start = pos + boundary.length + 2;
116             if (body[start] == '-') {
117                 break;
118             }
119             start += 2;
120             final int stop = RqMultipart.indexOf(body, boundary, start) - 2;
121             requests.add(this.make(body, start, stop - 2));
122             pos = stop;
123         }
124         this.map = RqMultipart.asMap(requests);
125     }
126 
127     /**
128      * Get single part.
129      * @param name Name of the part to get
130      * @return List of parts (can be empty)
131      */
132     public Iterable<Request> part(final String name) {
133         final List<Request> values = this.map
134             .get(name.toLowerCase(Locale.ENGLISH));
135         final Iterable<Request> iter;
136         if (values == null) {
137             iter = new VerboseIterable<Request>(
138                 Collections.<Request>emptyList(),
139                 new Sprintf(
140                     "there are no parts by name \"%s\" among %d others: %s",
141                     name, this.map.size(), this.map.keySet()
142                 )
143             );
144         } else {
145             iter = new VerboseIterable<Request>(
146                 values,
147                 new Sprintf(
148                     "there are just %d parts by name \"%s\"",
149                     values.size(), name
150                 )
151             );
152         }
153         return iter;
154     }
155 
156     /**
157      * Get all part names.
158      * @return All names
159      */
160     public Iterable<String> names() {
161         return this.map.keySet();
162     }
163 
164     /**
165      * Make a request.
166      * @param body Body
167      * @param start Start position
168      * @param stop Stop position
169      * @return Request
170      * @throws IOException If fails
171      */
172     private Request make(final byte[] body, final int start,
173         final int stop) throws IOException {
174         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
175         baos.write(this.head().iterator().next().getBytes());
176         baos.write("\r\n".getBytes());
177         baos.write(Arrays.copyOfRange(body, start, stop));
178         return new RqLive(new ByteArrayInputStream(baos.toByteArray()));
179     }
180 
181     /**
182      * Convert a list of requests to a map.
183      * @param reqs Requests
184      * @return Map of them
185      * @throws IOException If fails
186      */
187     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
188     private static ConcurrentMap<String, List<Request>> asMap(
189         final Collection<Request> reqs) throws IOException {
190         final ConcurrentMap<String, List<Request>> map =
191             new ConcurrentHashMap<String, List<Request>>(reqs.size());
192         for (final Request req : reqs) {
193             final String header = new RqHeaders(req)
194                 .header("Content-Disposition").iterator().next();
195             final Matcher matcher = RqMultipart.NAME.matcher(header);
196             if (!matcher.matches()) {
197                 throw new IOException(
198                     String.format(
199                         "\"name\" not found in Content-Disposition header: %s",
200                         header
201                     )
202                 );
203             }
204             final String name = matcher.group(1);
205             map.putIfAbsent(name, new LinkedList<Request>());
206             map.get(name).add(req);
207         }
208         return map;
209     }
210 
211     /**
212      * Find position of array inside another array.
213      * @param outer Big array
214      * @param inner Small array
215      * @param start Where to start searching
216      * @return Position
217      * @throws IOException If fails
218      */
219     private static int indexOf(final byte[] outer, final byte[] inner,
220         final int start) throws IOException {
221         for (int idx = start; idx < outer.length - inner.length; ++idx) {
222             boolean found = true;
223             for (int sub = 0; sub < inner.length; ++sub) {
224                 if (outer[idx + sub] != inner[sub]) {
225                     found = false;
226                     break;
227                 }
228             }
229             if (found) {
230                 return idx;
231             }
232         }
233         throw new IOException("closing boundary not found");
234     }
