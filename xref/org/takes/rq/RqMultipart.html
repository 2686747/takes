63  @SuppressWarnings("PMD.TooManyMethods")
64  public interface RqMultipart extends Request {
65

79      /**
80       * Request decorator, that decodes FORM data from
81       * {@code multipart/form-data} format (RFC 2045).
82       *
83       * <p>For {@code application/x-www-form-urlencoded}
84       * format use {@link org.takes.rq.RqForm}.
85       *
86       * <p>It is highly recommended to use {@link org.takes.rq.RqGreedy}
87       * decorator before passing request to this class.
88       *
89       * <p>The class is immutable and thread-safe.
90       *
91       * @author Yegor Bugayenko (yegor@teamed.io)
92       * @version $Id: 73d32cb4f885f9b230af219c70df0f0e982dc675 $
93       * @since 0.9
94       * @see <a href="http://www.w3.org/TR/html401/interact/forms.html">Forms in HTML</a>
95       * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
96       * @see org.takes.rq.RqGreedy
97       */
98      @EqualsAndHashCode(callSuper = true)
99      final class Base extends RqWrap implements RqMultipart {
100         /**
101          * Pattern to get boundary from header.
102          */
103         @SuppressWarnings("PMD.UnusedPrivateField")
104         private static final Pattern BOUNDARY = Pattern.compile(
105             ".*[^a-z]boundary=([^;]+).*"
106         );
107         /**
108          * Pattern to get name from header.
109          */
110         @SuppressWarnings("PMD.UnusedPrivateField")
111         private static final Pattern NAME = Pattern.compile(
112             ".*[^a-z]name=\"([^\"]+)\".*"
113         );
114         /**
115          * Map of params and values.
116          */
117         private final transient ConcurrentMap<String, List<Request>> map;
118         /**
119          * Internal buffer.
120          */
121         private final transient ByteBuffer buffer;
122         /**
123          * Origin request body.
124          */
125         private final transient ReadableByteChannel body;
126         /**
127          * Ctor.
128          * @param req Original request
129          * @throws IOException If fails
130          * @checkstyle ExecutableStatementCountCheck (2 lines)
131          */
132         public Base(final Request req) throws IOException {
133             super(req);
134             final InputStream stream = new RqLengthAware(req).body();
135             this.body = Channels.newChannel(stream);
136             this.buffer = ByteBuffer.allocate(
137                 // @checkstyle MagicNumberCheck (1 line)
138                 Math.min(8192, stream.available())
139             );
140             this.map = this.buildRequests(req);
141         }
142         @Override
143         public Iterable<Request> part(final CharSequence name) {
144             final List<Request> values = this.map
145                 .get(name.toString().toLowerCase(Locale.ENGLISH));
146             final Iterable<Request> iter;
147             if (values == null) {
148                 iter = new VerboseIterable<Request>(
149                     Collections.<Request>emptyList(),
150                     new Sprintf(
151                         "there are no parts by name \"%s\" among %d others: %s",
152                         name, this.map.size(), this.map.keySet()
153                     )
154                 );
155             } else {
156                 iter = new VerboseIterable<Request>(
157                     values,
158                     new Sprintf(
159                         "there are just %d parts by name \"%s\"",
160                         values.size(), name
161                     )
162                 );
163             }
164             return iter;
165         }
166         @Override
167         public Iterable<String> names() {
168             return this.map.keySet();
169         }
170         /**
171          * Build a request for each part of the origin request.
172          * @param req Origin request
173          * @return The requests map that use the part name as a map key
174          * @throws IOException If fails
175          */
176         private ConcurrentMap<String, List<Request>> buildRequests(
177             final Request req) throws IOException {
178             final String header = new RqHeaders.Smart(
179                 new RqHeaders.Base(req)
180             // @checkstyle MultipleStringLiteralsCheck (1 line)
181             ).single("Content-Type");
182             if (!header.toLowerCase(Locale.ENGLISH)
183                 .startsWith("multipart/form-data")) {
184                 throw new HttpException(
185                     HttpURLConnection.HTTP_BAD_REQUEST,
186                     String.format(
187                         // @checkstyle LineLength (1 line)
188                         "RqMultipart.Base can only parse multipart/form-data, while Content-Type specifies a different type: \"%s\"",
189                         header
190                     )
191                 );
192             }
193             final Matcher matcher = RqMultipart.Base.BOUNDARY.matcher(header);
194             if (!matcher.matches()) {
195                 throw new HttpException(
196                     HttpURLConnection.HTTP_BAD_REQUEST,
197                     String.format(
198                         // @checkstyle LineLength (1 line)
199                         "boundary is not specified in Content-Type header: \"%s\"",
200                         header
201                     )
202                 );
203             }
204             if (this.body.read(this.buffer) < 0) {
205                 throw new HttpException(
206                     HttpURLConnection.HTTP_BAD_REQUEST,
207                     "failed to read the request body"
208                 );
209             }
210             final byte[] boundary = String.format(
211                 "\r\n--%s", matcher.group(1)
212             ).getBytes();
213             this.buffer.flip();
214             this.buffer.position(boundary.length - 2);
215             final Collection<Request> requests = new LinkedList<Request>();
216             while (this.buffer.hasRemaining()) {
217                 final byte data = this.buffer.get();
218                 if (data == '-') {
219                     break;
220                 }
221                 this.buffer.position(this.buffer.position() + 1);
222                 requests.add(this.make(boundary));
223             }
224             return RqMultipart.Base.asMap(requests);
225         }
226         /**
227          * Make a request.
228          *  Scans the origin request until the boundary reached. Caches
229          *  the  content into a temporary file and returns it as a new request.
230          * @param boundary Boundary
231          * @return Request
232          * @throws IOException If fails
233          */
234         private Request make(final byte[] boundary) throws IOException {
235             final File file = File.createTempFile(
236                 RqMultipart.class.getName(), ".tmp"
237             );
238             file.deleteOnExit();
239             final FileChannel channel = new RandomAccessFile(
240                 file, "rw"
241             ).getChannel();
242             try {
243                 channel.write(
244                     ByteBuffer.wrap(this.head().iterator().next().getBytes())
245                 );
246                 // @checkstyle MultipleStringLiteralsCheck (1 line)
247                 channel.write(ByteBuffer.wrap("\r\n".getBytes()));
248                 this.copy(channel, boundary);
249             } finally {
250                 channel.close();
251             }
252             return new RqLive(
253                 new CapInputStream(
254                     new FileInputStream(file),
255                     file.length()
256                 )
257             );
258         }
259         /**
260          * Copy until boundary reached.
261          * @param target Output file channel
262          * @param boundary Boundary
263          * @throws IOException If fails
264          * @checkstyle ExecutableStatementCountCheck (2 lines)
265          */
266         private void copy(final WritableByteChannel target,
267             final byte[] boundary) throws IOException {
268             int match = 0;
269             boolean cont = true;
270             while (cont) {
271                 if (!this.buffer.hasRemaining()) {
272                     this.buffer.clear();
273                     for (int idx = 0; idx < match; ++idx) {
274                         this.buffer.put(boundary[idx]);
275                     }
276                     match = 0;
277                     if (this.body.read(this.buffer) == -1) {
278                         break;
279                     }
280                     this.buffer.flip();
281                 }
282                 final ByteBuffer btarget = this.buffer.slice();
283                 final int offset = this.buffer.position();
284                 btarget.limit(0);
285                 while (this.buffer.hasRemaining()) {
286                     final byte data = this.buffer.get();
287                     if (data == boundary[match]) {
288                         ++match;
289                         if (match == boundary.length) {
290                             cont = false;
291                             break;
292                         }
293                     } else {
294                         match = 0;
295                         btarget.limit(this.buffer.position() - offset);
296                     }
297                 }
298                 target.write(btarget);
299             }
300         }
301         /**
302          * Convert a list of requests to a map.
303          * @param reqs Requests
304          * @return Map of them
305          * @throws IOException If fails
306          */
307         @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
308         private static ConcurrentMap<String, List<Request>> asMap(
309             final Collection<Request> reqs) throws IOException {
310             final ConcurrentMap<String, List<Request>> map =
311                 new ConcurrentHashMap<String, List<Request>>(reqs.size());
312             for (final Request req : reqs) {
313                 final String header = new RqHeaders.Smart(
314                     new RqHeaders.Base(req)
315                 // @checkstyle MultipleStringLiteralsCheck (1 line)
316                 ).single("Content-Disposition");
317                 final Matcher matcher = RqMultipart.Base.NAME.matcher(header);
318                 if (!matcher.matches()) {
319                     throw new HttpException(
320                         HttpURLConnection.HTTP_BAD_REQUEST,
321                         String.format(
322                             // @checkstyle LineLength (1 line)
323                             "\"name\" not found in Content-Disposition header: %s",
324                             header
325                         )
326                     );
327                 }
328                 final String name = matcher.group(1);
329                 map.putIfAbsent(name, new LinkedList<Request>());
330                 map.get(name).add(req);
331             }
332             return map;
333         }
334     }

336     /**
337      * Smart decorator.
338      * @since 0.15
339      */
340     final class Smart implements RqMultipart {
341         /**
342          * Original request.
343          */
344         private final transient RqMultipart origin;
345         /**
346          * Ctor.
347          * @param req Original
348          */
349         public Smart(final RqMultipart req) {
350             this.origin = req;
351         }
352         /**
353          * Get single part.
354          * @param name Name of the part to get
355          * @return Part
356          * @throws HttpException If fails
357          */
358         public Request single(final CharSequence name) throws HttpException {
359             final Iterator<Request> parts = this.part(name).iterator();
360             if (!parts.hasNext()) {
361                 throw new HttpException(
362                     HttpURLConnection.HTTP_BAD_REQUEST,
363                     String.format(
364                         "form param \"%s\" is mandatory", name
365                     )
366                 );
367             }
368             return parts.next();
369         }
370         @Override
371         public Iterable<Request> part(final CharSequence name) {
372             return this.origin.part(name);
373         }
374         @Override
375         public Iterable<String> names() {
376             return this.origin.names();
377         }
378         @Override
379         public Iterable<String> head() throws IOException {
380             return this.origin.head();
381         }
382         @Override
383         public InputStream body() throws IOException {
384             return this.origin.body();
385         }
386     }

388     /**
389      * Fake decorator.
390      * @since 0.16
391      */
392     final class Fake implements RqMultipart {
393         /**
394          * Fake boundary constant.
395          */
396         private static final String BOUNDARY = "AaB02x";
397         /**
398          * Carriage return constant.
399          */
400         private static final String CRLF = "\r\n";
401         /**
402          * Fake multipart request.
403          */
404         private final RqMultipart fake;
405         /**
406          * Fake ctor.
407          * @param req Fake request header holder
408          * @param dispositions Fake request body parts
409          * @throws IOException If fails
410          */
411         public Fake(final Request req, final Request... dispositions)
412             throws IOException {
413             this.fake = new RqMultipart.Base(
414                 new Request() {
415                     @Override
416                     public Iterable<String> head() throws IOException {
417                         return new RqWithHeader(
418                             req,
419                             // @checkstyle MultipleStringLiteralsCheck (1 line)
420                             "Content-Type",
421                             String.format(
422                                 "multipart/form-data; boundary=%s",
423                                 RqMultipart.Fake.BOUNDARY
424                             )
425                         ).head();
426                     }
427                     @Override
428                     public InputStream body() throws IOException {
429                         return RqMultipart.Fake.fakeBody(dispositions);
430                     }
431                 }
432             );
433         }
434         @Override
435         public Iterable<Request> part(final CharSequence name) {
436             return this.fake.part(name);
437         }
438         @Override
439         public Iterable<String> names() {
440             return this.fake.names();
441         }
442         @Override
443         public Iterable<String> head() throws IOException {
444             return this.fake.head();
445         }
446         @Override
447         public InputStream body() throws IOException {
448             return this.fake.body();
449         }
450         /**
451          * Fake body creator.
452          * @param dispositions Fake request body parts
453          * @return InputStream of given dispositions
454          * @throws IOException If fails
455          */
456         @SuppressWarnings("PMD.InsufficientStringBufferDeclaration")
457         private static InputStream fakeBody(final Request... dispositions)
458             throws IOException {
459             final StringBuilder builder = new StringBuilder();
460             for (final Request each : dispositions) {
461                 builder.append(String.format("--%s", BOUNDARY))
462                     .append(CRLF)
463                     // @checkstyle MultipleStringLiteralsCheck (1 line)
464                     .append("Content-Disposition: ")
465                     .append(
466                         new RqHeaders.Smart(
467                             new RqHeaders.Base(each)
468                         // @checkstyle MultipleStringLiteralsCheck (1 line)
469                         ).single("Content-Disposition")
470                     ).append(CRLF);
471                 final String body = new RqPrint(each).printBody();
472                 if (!(CRLF.equals(body) || "".equals(body))) {
473                     builder.append(CRLF).append(body).append(CRLF);
474                 }
475             }
476             builder.append("Content-Transfer-Encoding: utf-8").append(CRLF)
477                 .append(String.format("--%s--", BOUNDARY));
478             return new ByteArrayInputStream(
479                 builder.toString().getBytes()
480             );
481         }
482     }
