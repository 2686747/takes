24  package org.takes.rs.xe;
25  
26  import java.io.IOException;
27  import java.util.Iterator;
28  import lombok.EqualsAndHashCode;
29  
30  /**
31   * Iterable to transform an iterable of some objects
32   * into an iterable of Xembly sources.
33   *
34   * <p>Use this class to create a collection of
35   * {@link org.takes.rs.xe.XeSource} objects and pass them to,
36   * for example, {@link org.takes.rs.xe.XeAppend}:
37   *
38   * <pre> return new RsXembly(
39   *   new XeAppend(
40   *     "books",
41   *     new XeTransform&lt;Book&gt;(
42   *       this.database.books(),
43   *       new XeTransform.Func&lt;Book&gt;() {
44   *         &#64;Override
45   *         public XeSource transform(final Book book) {
46   *           return new XeAppend(
47   *             "book",
48   *             new XeDirectives(
49   *               new Directives()
50   *                 .add("book")
51   *                 .attr("isbn", book.isbn());
52   *             )
53   *           )
54   *         }
55   *       }
56   *     )
57   *   )
58   * );</pre>
59   *
60   * <p>The class is immutable and thread-safe.
61   *
62   * @author Yegor Bugayenko (yegor@teamed.io)
63   * @version $Id$
64   * @since 0.1
65   */
66  @EqualsAndHashCode(of = { "objects", "func" })
67  public final class XeTransform<T> implements Iterable<XeSource> {
68  
69      /**
70       * Iterable of objects.
71       */
72      private final transient Iterable<T> objects;
73  
74      /**
75       * Function to use for mapping.
76       */
77      private final transient XeTransform.Func<T> func;
78  
79      /**
80       * Ctor.
81       * @param list List of objects
82       * @param fnc Function
83       */
84      public XeTransform(final Iterable<T> list, final XeTransform.Func<T> fnc) {
85          this.objects = list;
86          this.func = fnc;
87      }
88  
89      @Override
90      public Iterator<XeSource> iterator() {
91          final Iterator<T> origin = this.objects.iterator();
92          return new Iterator<XeSource>() {
93              @Override
94              public boolean hasNext() {
95                  return origin.hasNext();
96              }
97              @Override
98              public XeSource next() {
99                  try {
100                     return XeTransform.this.func.transform(origin.next());
101                 } catch (final IOException ex) {
102                     throw new IllegalStateException(ex);
103                 }
104             }
105             @Override
106             public void remove() {
107                 origin.remove();
108             }
109         };
110     }
111 
112     /**
113      * Function to map them.
114      */
115     public interface Func<T> {
116         /**
117          * Transform an object.
118          * @param obj Object
119          * @return Xembly source
120          * @throws IOException If fails
121          */
122         XeSource transform(T obj) throws IOException;
123     }
124 }
