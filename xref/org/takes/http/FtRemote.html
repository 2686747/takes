24  package org.takes.http;
25  
26  import java.io.IOException;
27  import java.net.ServerSocket;
28  import java.net.URI;
29  import java.util.Collection;
30  import java.util.concurrent.ConcurrentSkipListSet;
31  import java.util.concurrent.CountDownLatch;
32  import java.util.concurrent.TimeUnit;
33  import java.util.concurrent.atomic.AtomicBoolean;
34  import lombok.EqualsAndHashCode;
35  import org.takes.Takes;
36  
37  /**
38   * Front remote control.
39   *
40   * <p>The class is immutable and thread-safe.
41   *
42   * @author Yegor Bugayenko (yegor@teamed.io)
43   * @version $Id$
44   * @since 0.1
45   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
46   */
47  @SuppressWarnings("PMD.DoNotUseThreads")
48  @EqualsAndHashCode(of = { "back", "port" })
49  public final class FtRemote implements Front {
50  
51      /**
52       * Already assigned ports.
53       */
54      private static final Collection<Integer> PORTS =
55          new ConcurrentSkipListSet<Integer>();
56  
57      /**
58       * Back.
59       */
60      private final transient Back back;
61  
62      /**
63       * Port.
64       */
65      private final transient int port;
66  
67      /**
68       * Ctor.
69       * @param tks Takes
70       * @throws IOException If fails
71       */
72      public FtRemote(final Takes tks) throws IOException {
73          this(new BkParallel(new BkBasic(tks)));
74      }
75  
76      /**
77       * Ctor.
78       * @param bck Back
79       * @throws IOException If fails
80       */
81      public FtRemote(final Back bck) throws IOException {
82          this.back = bck;
83          this.port = FtRemote.allocate();
84      }
85  
86      @Override
87      public void start(final Exit exit) throws IOException {
88          new FtBasic(this.back, this.port).start(exit);
89      }
90  
91      /**
92       * Execute this script against a running front.
93       * @param script Script to run
94       * @throws IOException If fails
95       */
96      public void exec(final FtRemote.Script script) throws IOException {
97          final AtomicBoolean exit = new AtomicBoolean();
98          final CountDownLatch latch = new CountDownLatch(1);
99          final Thread thread = new Thread(
100             new Runnable() {
101                 @Override
102                 public void run() {
103                     try {
104                         FtRemote.this.start(
105                             new Exit() {
106                                 @Override
107                                 public boolean ready() {
108                                     latch.countDown();
109                                     return exit.get();
110                                 }
111                             }
112                         );
113                     } catch (final IOException ex) {
114                         throw new IllegalStateException(ex);
115                     }
116                 }
117             }
118         );
119         thread.start();
120         try {
121             // @checkstyle MagicNumber (1 line)
122             latch.await(10L, TimeUnit.SECONDS);
123         } catch (final InterruptedException ex) {
124             Thread.currentThread().interrupt();
125             throw new IllegalStateException(ex);
126         }
127         script.exec(
128             URI.create(
129                 String.format("http://localhost:%d", this.port)
130             )
131         );
132         exit.set(true);
133         try {
134             thread.join();
135         } catch (final InterruptedException ex) {
136             Thread.currentThread().interrupt();
137             throw new IllegalStateException(ex);
138         }
139         FtRemote.PORTS.remove(this.port);
140     }
141 
142     /**
143      * Allocate a new random TCP port.
144      * @return TCP port
145      * @throws IOException If fails
146      */
147     private static int allocate() throws IOException {
148         synchronized (FtRemote.PORTS) {
149             int attempts = 0;
150             int port;
151             do {
152                 port = FtRemote.random();
153                 ++attempts;
154                 // @checkstyle MagicNumber (1 line)
155                 if (attempts > 100) {
156                     throw new IllegalStateException(
157                         String.format(
158                             "failed to allocate TCP port after %d attempts",
159                             attempts
160                         )
161                     );
162                 }
163             } while (FtRemote.PORTS.contains(port));
164             return port;
165         }
166     }
167 
168     /**
169      * Allocate a new random TCP port.
170      * @return TCP port
171      * @throws IOException If fails
172      */
173     private static int random() throws IOException {
174         final ServerSocket socket = new ServerSocket(0);
175         try {
176             socket.setReuseAddress(true);
177             return socket.getLocalPort();
178         } finally {
179             socket.close();
180         }
181     }
182 
183     /**
184      * Script to execute.
185      */
186     public interface Script {
187         /**
188          * Execute it against this URI.
189          * @param home URI of the running front
190          * @throws IOException If fails
191          */
192         void exec(URI home) throws IOException;
193     }
194 
195 }
